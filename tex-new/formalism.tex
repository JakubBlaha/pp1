
\section{Introduction}

This document defines a logic-agnostic formal base for expressing software requirements and test cases. It is intended as a close-to-natural-language intermediate representation that an LLM (or a human) can read and produce: its constructs, expressions, and predicates are named after the vocabulary already used when describing requirements informally, while each name is tied to a precise mathematical definition. Its purpose is to enable mechanical detection of coverage gaps: given a set of requirements and a set of test cases formalised over a shared set of entities, a checker determines which requirement conditions are not exercised by any test.

The base is organised into several layers, each building on the ones below it. At the bottom, a time set and a set of \emph{entities} (signals, storage, channels, states, events, and so on) describe the kinds of things present in the system, and a \emph{trace} assigns to each time point a valuation mapping entities to their current values. Built on top of traces, \emph{expressions} compute values and \emph{predicates} evaluate to truth values at a given point in time. \emph{Operations} are the actions a test case fires to drive a trace, each producing an effect predicate that states what must hold as a result of the firing. \emph{Temporal constructs} then lift point-in-time predicates into statements about the whole trace by quantifying over time.

A requirement and a test case are both built from these same layers over the same entities, drawing their predicates from the same space; this shared vocabulary is what allows a checker to determine which conditions of a requirement a given test case exercises. The catalogues of entity kinds and operations are kept open to extension, so that supporting a new kind of entity or action requires only widening the relevant catalogue while every existing construct continues to work unchanged.

\section{The primitives}

These primitive sets serve as the base that the rest of the formalism builds on. The primitive sets are the following:
\begin{itemize}
    \item $T_D$, $T_C$ -- the \textit{discrete time} and \textit{continuous time} sets respectively.
    \item $\mathit{ID}_E$ -- the set of entity identifiers.
    \item $V_{\text{base}}$ -- the base values set: booleans, reals, time points, and intervals.
    \item $V_{\text{nonset}}$, $V$ -- defined in Section~\ref{sec:entities} once entities are available; $V_{\text{nonset}}$ is the union of base values and entities; $V$ additionally includes finite sets whose elements are drawn from $V_{\text{nonset}}$.
    \item $\mathit{EventTypeLabel}$ -- the set of event type labels; defined below.
\end{itemize}

\begin{definition}[Discrete time set]\label{def:time_discrete}
    The \emph{discrete time set} $T_D$ is totally ordered and countable, extended with a greatest element $\infty$:
    \[
        T_D = \mathbb{N} \cup \{\infty\}
    \]
    The ordering is the usual order on $\mathbb{N}$ with $\forall n \in \mathbb{N}: n < \infty$.
\end{definition}

\begin{definition}[Continuous time set]\label{def:time_continuous}
    The \emph{continuous time set} $T_C$ is totally ordered and uncountable, extended with a greatest element $\infty$:
    \[
        T_C = \mathbb{R}_{\geq 0} \cup \{\infty\}
    \]
    The ordering is the usual order on $\mathbb{R}_{\geq 0}$ with $\forall x \in \mathbb{R}_{\geq 0}: x < \infty$.
\end{definition}

We say that $T$ is an infinite totally ordered set; the time set can have \textit{discrete} or \textit{continuous} flavour; In the scope of a single requirement, $T$ is equal either to $T_D$ or $T_C$.

\begin{definition}[Duration]\label{def:duration}
    A \emph{duration} is a non-negative element of the time set, representing the length of a time span:
    \[
        \Delta = T
    \]
    Thus $\Delta = \mathbb{N}$ for discrete time and $\Delta = \mathbb{R}_{\geq 0}$ for continuous time.
\end{definition}

\begin{definition}[Interval]\label{def:interval}
    An \emph{interval} is a tuple of a start time point, a duration, and two endpoint-openness flags:
    \[
        \mathcal{I} = T \times \Delta \times \mathbb{B} \times \mathbb{B}
    \]
    An interval $(t, d, \ell, r) \in \mathcal{I}$ represents a time span with start $t$, end $t+d$, left-open flag $\ell$, and right-open flag $r$:
    \begin{center}
        \begin{tabular}{ccc}
            \toprule
            $\ell$           & $r$              & represented span \\
            \midrule
            $\mathit{false}$ & $\mathit{false}$ & $[t,\; t+d]$     \\
            $\mathit{true}$  & $\mathit{false}$ & $(t,\; t+d]$     \\
            $\mathit{false}$ & $\mathit{true}$  & $[t,\; t+d)$     \\
            $\mathit{true}$  & $\mathit{true}$  & $(t,\; t+d)$     \\
            \bottomrule
        \end{tabular}
    \end{center}
\end{definition}

\begin{definition}[Entity identifier set]\label{def:entity_id_set}
    The \emph{entity identifier set} $\mathit{ID}_E$ is a finite nonempty set of unique abstract tokens called \emph{entity identifiers}.
\end{definition}

\begin{definition}[Base values]\label{def:base_values}
    The \emph{base values set} $V_{\text{base}}$ is:
    \[
        V_{\text{base}} = \mathbb{B} \cup \mathbb{R} \cup T \cup \mathcal{I}
    \]
    where $\mathbb{B} = \{\mathit{true}, \mathit{false}\}$ is the set of booleans, $\mathbb{R}$ is the set of real numbers, $T$ is the time set, and $\mathcal{I}$ is the set of intervals.
\end{definition}

\begin{definition}[Event type label set]\label{def:event_type_label_set}
    The \emph{event type label set} $\mathit{EventTypeLabel}$ is a finite set of abstract symbols:
    \begin{gather*}
        \mathit{EventTypeLabel} = \{\,
        \mathit{generic},\;
        \mathit{calculated},\;
        \mathit{written},\;
        \mathit{read},\;
        \mathit{entered},\;
        \mathit{transmitted},\; \\
        \mathit{received},\;
        \mathit{inserted},\;
        \mathit{removed},\;
        \mathit{cleared}
        \,\}
    \end{gather*}
\end{definition}

\section{Entities and entity types}\label{sec:entities}

An entity type is an item belonging to a set of entity types. There is a number of defined entity types and this set is open to extension.

\begin{definition}[Entity type set]\label{def:entity_type_set}
    \[
        T_\text{E} = \{ \text{Signal}, \text{Storage}, \text{Event}, \text{EventTrigger}, \text{Channel}, \text{State}, \text{Value}, \text{Abstract}, \text{Set}\}
    \]
\end{definition}

Entity type modifiers are properties which can be attached to an entity of a particular type.
A modifier is either a \emph{flag}, whose presence alone carries meaning, or a \emph{parameterised modifier}, which additionally carries a value in $\mathit{ModVal}$.

\begin{definition}[Modifier value set]\label{def:mod_val}
    The \emph{modifier value set} is:
    \[
        \mathit{ModVal} = V_{\text{base}} \cup \mathit{ID}_E \cup \mathit{EventTypeLabel}
    \]
\end{definition}

Each entity type is compatible with only a subset of the modifier keys.
Therefore we define the \textit{entity type modifiers function}.

\begin{definition}[Entity type modifiers function]\label{def:entity_type_modifiers_fn}
    $M_\text{E}: T_E \rightarrow \mathcal{P}(\mathit{ModKey})$ maps each entity type to its set of permitted modifier keys,
    where $\mathit{ModKey}$ is a finite nonempty set of modifier keys.
    The permitted modifier keys and their expected value types are:
    \begin{center}
        \begin{tabular}{lll}
            \toprule
            \textbf{Entity type} & \textbf{Modifier key} & \textbf{Value type}       \\
            \midrule
            \text{Storage}       & $\mathit{address}$    & $\mathbb{N}$              \\
            \text{Event}         & $\mathit{target}$     & $\mathit{ID}_E$           \\
            \text{Event}         & $\mathit{type}$       & $\mathit{EventTypeLabel}$ \\
            \bottomrule
        \end{tabular}
    \end{center}
    Every \texttt{Event} entity is expected to carry both modifiers: user-declared events set $\mathit{target}$ to the identifier of their \texttt{EventTrigger} and $\mathit{type}$ to $\mathit{generic}$; predefined events set $\mathit{target}$ to their parent entity and $\mathit{type}$ to the appropriate event type.

    \textit{Note:} At most one \texttt{Event} entity with a given $(\mathit{target}, \mathit{type})$ pair should exist in $E$ for a given requirement.
\end{definition}

An entity is something that is referenced in a requirement and corresponding test cases.
Formally, an entity is defined as a three-item tuple.

\begin{definition}[Entity]\label{def:entity}
    \[
        e = (id_e, t_e, \mu_e)
    \]

    , where
    \begin{itemize}
        \item $id_e$ is the unique identifier of the entity; $id_e \in \mathit{ID}_E$
        \item $t_e$ is the type of the entity; $t_e \in T_E$
        \item $\mu_e$ is the modifier assignment function of the entity; $\mu_e : \mathit{ModKey} \rightharpoonup \mathit{ModVal}$, $\mathrm{dom}(\mu_e) \subseteq M_E(t_e)$ \footnote{$\mathrm{dom}(\mu_e)$ is a subset of the set of modifier keys compatible with the entity type $t_e$. It is not required that a value is defined for each modifier key compatible with $t_e$.}
    \end{itemize}
    No two entities in $E$ share the same identifier: $\forall e, e' \in E: e \neq e' \Rightarrow id_e \neq id_{e'}$.
    We write $E$ for the set of all entities. The set $E$ is finite and nonempty.\footnote{The set $E$ is finite in the scope of a single requirement.}
\end{definition}

\begin{definition}[Non-set values]\label{def:nonset_values}
    The \emph{non-set values} $V_{\text{nonset}}$ is the union of base values and entities:
    \[
        V_{\text{nonset}} = V_{\text{base}} \cup E
    \]
\end{definition}

\begin{definition}[Values]\label{def:values}
    The \emph{values set} $V$ is:
    \[
        V = V_{\text{nonset}} \cup \mathcal{P}_{\text{fin}}(V_{\text{nonset}})
    \]
    where $\mathcal{P}_{\text{fin}}(V_{\text{nonset}})$ is the set of all finite subsets of $V_{\text{nonset}}$.
    Set values are flat: they may contain only non-set values.
\end{definition}

\section{Valuations and traces}

\begin{definition}[Valuation]\label{def:valuation}
    A valuation is a partial function $\sigma : E \rightharpoonup V$ assigning a value to a subset of entities at each time point\footnote{For instance a register entity can have its value undefined at time points preceeding a write to this register.}; $\sigma(e)$ is undefined when the entity has no defined value at that time point.
    For entities of type \texttt{Event}, $\sigma(e)$ is always defined and $\sigma(e) \in \{\mathit{true}, \mathit{false}\}$,
    where $\sigma(e) = \mathit{true}$ iff the event occurs at that time point.
    For entities of type \texttt{Set}, $\sigma(e)$, when defined, is a finite subset of $V_{\text{nonset}}$, i.e.\ $\sigma(e) \in \mathcal{P}_{\text{fin}}(V_{\text{nonset}})$.
\end{definition}

We write $\Sigma$ for the infinite set of all valuations.

\begin{definition}[Trace]\label{def:trace}
    A trace is a function $\rho : T \to \Sigma$. We write $\rho(t)(e)$ for the value of entity $e$ at time $t$.
\end{definition}

We write $\Sigma^T$ for the set of all traces, i.e.\ the set of all functions from $T$ to $\Sigma$.

\footnote{This is standard function space notation: $\Sigma^T = \{\, \rho \mid \rho : T \to \Sigma \,\}$.}

\section{Expressions and predicates}

All expression functions defined in this section evaluate to some value $v \in V$.
We write $\mathbf{Expr}$ for the infinite set of all expressions.
\subsection{Expressions}\label{subsec:expressions}

\subsubsection{Constants}\label{subsec:constants}

\begin{definition}[Constant]\label{def:constant}
    Any value $c \in V$ may appear as a constant expression.
\end{definition}

\subsubsection{Arithmetic operators}\label{subsec:arithmetic}

\begin{definition}[Arithmetic operators]\label{def:arithmetic}
    The standard arithmetic operators $+$, $-$, $\times$, $/$ are defined on numeric values ($\mathbb{R}$ and its subsets) in the usual sense and can be used to construct expressions.
\end{definition}

\subsubsection{Time functions}\label{subsec:time}

The following functions depend only on time points or intervals; they do not depend on a trace.

\begin{definition}[Current time]\label{def:now}
    $\mathit{Now} \in T$ returns the ambient time point $t$:
    \[
        \mathit{Now} \;=\; t
    \]
\end{definition}

The following two functions are defined for \textbf{discrete time only} ($T = T_D = \mathbb{N}$).

\begin{definition}[]\label{def:prev}
    $\mathit{Prev} : T_D \rightharpoonup T_D$ returns the preceding time step:
    \[
        \mathit{Prev}(t) \;=\; t - 1
    \]
    The function is partial: it is undefined at $t = 0$ since $t - 1 \notin T_D$.
\end{definition}

\begin{definition}[]\label{def:next}
    $\mathit{Next} : T_D \rightarrow T_D$ returns the following time step:
    \[
        \mathit{Next}(t) \;=\; t + 1
    \]
\end{definition}

\begin{definition}[]\label{def:reltime}
    $\mathit{RelTime} : T_D \times \mathbb{Z} \rightharpoonup T_D$ shifts a time point by $n$ steps, where $n \in \mathbb{Z}$ may be positive or negative:
    \[
        \mathit{RelTime}(t,\, n) \;=\; t + n
    \]
    The function is partial: it is undefined when $t + n < 0$ since $t + n \notin T_D$.
    Note that $\mathit{Prev}(t) = \mathit{RelTime}(t, -1)$ and $\mathit{Next}(t) = \mathit{RelTime}(t, 1)$.
\end{definition}

The following functions are defined for both time flavours.

\begin{definition}[]\label{def:diff}
    $\mathit{Diff} : T \times T \rightarrow \Delta$ returns the duration between two time points:
    \[
        \mathit{Diff}(t_1,\, t_2) \;=\; |t_1 - t_2|
    \]
\end{definition}

\begin{definition}[Interval constructors]\label{def:mk_interval}
    Four constructors build intervals from a start and end time point, differing only in endpoint openness.
    All are partial: undefined when $t_1 > t_2$.
    \begin{align*}
        \mathit{MkIntervalCC}(t_1,\, t_2) & \;=\; (t_1,\; t_2 - t_1,\; \mathit{false},\; \mathit{false}) &  & [t_1,\, t_2] \\
        \mathit{MkIntervalOC}(t_1,\, t_2) & \;=\; (t_1,\; t_2 - t_1,\; \mathit{true},\;  \mathit{false}) &  & (t_1,\, t_2] \\
        \mathit{MkIntervalCO}(t_1,\, t_2) & \;=\; (t_1,\; t_2 - t_1,\; \mathit{false},\; \mathit{true})  &  & [t_1,\, t_2) \\
        \mathit{MkIntervalOO}(t_1,\, t_2) & \;=\; (t_1,\; t_2 - t_1,\; \mathit{true},\;  \mathit{true})  &  & (t_1,\, t_2)
    \end{align*}
\end{definition}

\begin{definition}[Since-zero interval]\label{def:since_zero}
    $\mathit{SinceZero} : T \rightarrow \mathcal{I}$ constructs an interval from time zero to a given time point $t$:
    \[
        \mathit{SinceZero}(t) \;=\; (0,\; t)
    \]
\end{definition}

\begin{definition}[Interval start]\label{def:interval_start}
    $\mathit{Start} : \mathcal{I} \rightarrow T$ returns the start time point of an interval:
    \[
        \mathit{Start}((t,\, d,\, \ell,\, r)) \;=\; t
    \]
\end{definition}

\begin{definition}[Interval end]\label{def:interval_end}
    $\mathit{End} : \mathcal{I} \rightarrow T$ returns the end time point of an interval:
    \[
        \mathit{End}((t,\, d,\, \ell,\, r)) \;=\; t + d
    \]
\end{definition}

\begin{definition}[Interval duration]\label{def:interval_duration}
    $\mathit{Duration} : \mathcal{I} \rightarrow \Delta$ returns the duration of an interval:
    \[
        \mathit{Duration}((t,\, d,\, \ell,\, r)) \;=\; d
    \]
\end{definition}

\begin{definition}[Interval endpoint openness]\label{def:interval_open}
    $\mathit{IsLeftOpen},\, \mathit{IsRightOpen} : \mathcal{I} \rightarrow \mathbb{B}$ return the endpoint-openness flags:
    \[
        \mathit{IsLeftOpen}((t,\, d,\, \ell,\, r)) \;=\; \ell
        \qquad
        \mathit{IsRightOpen}((t,\, d,\, \ell,\, r)) \;=\; r
    \]
\end{definition}

\begin{definition}[Time point interval membership]\label{def:in_interval}
    $\mathit{InInterval} : T \times \mathcal{I} \rightarrow \mathbb{B}$ holds when time point $s$ falls within interval $i$:
    \begin{align*}
        \mathit{InInterval}(s,\, i) \;\Leftrightarrow\;
         & \bigl(\mathit{IsLeftOpen}(i) \Rightarrow \mathit{Start}(i) < s\bigr)         \\
        \land\;
         & \bigl(\lnot\mathit{IsLeftOpen}(i) \Rightarrow \mathit{Start}(i) \leq s\bigr) \\
        \land\;
         & \bigl(\mathit{IsRightOpen}(i) \Rightarrow s < \mathit{End}(i)\bigr)          \\
        \land\;
         & \bigl(\lnot\mathit{IsRightOpen}(i) \Rightarrow s \leq \mathit{End}(i)\bigr)
    \end{align*}
\end{definition}

\subsubsection{Entity attribute accessors}\label{subsec:entity_attrs}

These expressions return static properties of entities and do not depend on the trace or time.

\begin{definition}[Address]\label{def:addr}
    For a \texttt{Storage} entity $e \in E$ carrying an address modifier, $\mathit{Addr} : E \rightarrow V$ returns the static address value:
    \[
        \mathit{Addr}(e) \;=\; \mu_e(\mathit{address})
    \]
\end{definition}

\subsubsection{Trace functions}\label{subsec:trace}

The following functions additionally take a trace $\rho \in \Sigma^T$ and produce values that reflect the state of the system over time.

\begin{definition}[]\label{def:expr_val}
    For a trace $\rho \in \Sigma^T$, the partial function $\mathit{Val}_\rho: E \times T \rightharpoonup V$ denotes the value of entity $e \in E$ at time $t \in T$:
    \[
        \mathit{Val}_\rho(e,\, t) \;=\; \rho(t)(e)
    \]
    The function is partial: $\mathit{Val}_\rho(e,\, t)$ is undefined when $e \notin \mathrm{dom}(\rho(t))$.
\end{definition}

\begin{definition}[Value before time point]\label{def:val_before}
    For a trace $\rho \in \Sigma^T$, the partial function $\mathit{ValBefore}_\rho : E \times T \rightharpoonup V$ is the left limit of the value of $e$ at $t$:
    \[
        \mathit{ValBefore}_\rho(e,\, t) \;=\; \lim_{t' \to t^-} \mathit{Val}_\rho(e,\, t')
    \]
    The function is partial: it is undefined when the left limit does not exist, i.e.\ when no operation has fired on $e$ before $t$.
\end{definition}

\begin{definition}[Event occurrence count]\label{def:evt_occ_count}
    For a trace $\rho \in \Sigma^T$ and an entity $ev \in E$ of type \texttt{Event},
    $\mathit{EvtOccCount}_\rho : E \times \mathcal{I} \rightarrow \mathbb{N}$ returns the number of occurrences of $ev$ within an interval $i \in \mathcal{I}$:
    \[
        \mathit{EvtOccCount}_\rho(ev,\, i) \;=\; \bigl|\{\, s \in T \mid \mathit{InInterval}(s,\, i) \;\land\; \mathit{Val}_\rho(ev,\, s) = \mathit{true} \,\}\bigr|
    \]
\end{definition}

\begin{definition}[Last occurrences function]\label{def:last_occ}
    For a trace $\rho \in \Sigma^T$, an entity $ev \in E$ of type \texttt{Event}, a time point $t \in T$, and $n \in \mathbb{N}^+$,
    let $t_1 \leq t_2 \leq \cdots \leq t_n$ be the $n$ most recent occurrence times of $ev$ up to $t$, i.e.\ the $n$ largest elements of $\{\, s \in T \mid s \leq t \;\land\; \mathit{Val}_\rho(ev,\, s) = \mathit{true} \,\}$ sorted in ascending order.
    Then $\mathit{LastOcc}_\rho : E \times T \times \mathbb{N}^+ \rightharpoonup \mathcal{I}$ is defined as:
    \[
        \mathit{LastOcc}_\rho(ev,\, t,\, n) \;=\; \mathit{MkIntervalCC}(t_1,\, t_n)
    \]
    The function is partial: it is undefined when fewer than $n$ occurrences of $ev$ exist up to $t$.

    \textit{Note:} The time of a single occurrence is obtained via $\mathit{LastOcc}_\rho(ev, t, 1)$, which returns an interval of duration $0$. The occurrence time is then $\mathit{Start}(\mathit{LastOcc}_\rho(ev, t, 1))$.
\end{definition}

\begin{definition}[First occurrences function]\label{def:first_occ}
    For a trace $\rho \in \Sigma^T$, an entity $ev \in E$ of type \texttt{Event}, a time point $t \in T$, and $n \in \mathbb{N}^+$,
    let $t_1 \leq t_2 \leq \cdots \leq t_n$ be the $n$ earliest occurrence times of $ev$ at or after $t$, i.e.\ the $n$ smallest elements of $\{\, s \in T \mid s \geq t \;\land\; \mathit{Val}_\rho(ev,\, s) = \mathit{true} \,\}$ sorted in ascending order.
    Then $\mathit{FirstOcc}_\rho : E \times T \times \mathbb{N}^+ \rightharpoonup \mathcal{I}$ is defined as:
    \[
        \mathit{FirstOcc}_\rho(ev,\, t,\, n) \;=\; \mathit{MkIntervalCC}(t_1,\, t_n)
    \]
    The function is partial: it is undefined when fewer than $n$ occurrences of $ev$ exist at or after $t$.

    \textit{Note:} The time of the first occurrence is $\mathit{Start}(\mathit{FirstOcc}_\rho(ev, t, 1))$.
\end{definition}

\begin{definition}[Maximum value function]\label{def:max_val}
    For a trace $\rho \in \Sigma^T$ and an entity $e \in E$ with an ordered value domain,
    $\mathit{MaxVal}_\rho : E \times \mathcal{I} \rightarrow V$ returns the maximum value of $e$ over an interval $i \in \mathcal{I}$:
    \[
        \mathit{MaxVal}_\rho(e,\, i) \;=\; \sup\{\, \mathit{Val}_\rho(e,\, s) \mid s \in T,\; \mathit{Start}(i) \leq s \leq \mathit{End}(i) \,\}
    \]
\end{definition}

\begin{definition}[Minimum value function]\label{def:min_val}
    For a trace $\rho \in \Sigma^T$ and an entity $e \in E$ with an ordered value domain,
    $\mathit{MinVal}_\rho : E \times \mathcal{I} \rightarrow V$ returns the minimum value of $e$ over an interval $i \in \mathcal{I}$:
    \[
        \mathit{MinVal}_\rho(e,\, i) \;=\; \inf\{\, \mathit{Val}_\rho(e,\, s) \mid s \in T,\; \mathit{Start}(i) \leq s \leq \mathit{End}(i) \,\}
    \]
\end{definition}

\subsubsection{Set functions}\label{subsec:set_fns}

These functions operate on set values and do not depend on a trace.

\begin{definition}[Set size]\label{def:size}
    $\mathit{Size} : \mathcal{P}_{\text{fin}}(V_{\text{nonset}}) \rightarrow \mathbb{N}$ returns the cardinality of a set:
    \[
        \mathit{Size}(s) \;=\; |s|
    \]
\end{definition}

\begin{definition}[Set filter]\label{def:filter}
    For a set $s \in \mathcal{P}_{\text{fin}}(V_{\text{nonset}})$ and a predicate $P$ parameterised by $x \in V_{\text{nonset}}$,
    $\mathit{Filter} : \mathcal{P}_{\text{fin}}(V_{\text{nonset}}) \times \mathbf{Pred} \rightarrow \mathcal{P}_{\text{fin}}(V_{\text{nonset}})$ returns the subset of elements satisfying $P$:
    \[
        \mathit{Filter}(s,\, P) \;=\; \{\, x \in s \mid P(x) \,\}
    \]
\end{definition}

\subsection{Predicates}\label{subsec:predicates}

A predicate is a function that evaluates to a value in $\mathbb{B}$.
We write $\mathbf{Pred}$ for the infinite set of all predicates.

\begin{definition}[Boolean constants]\label{def:pred_bool}
    $\top \in \mathbf{Pred}$ and $\bot \in \mathbf{Pred}$ are the always-true and always-false predicates respectively.
\end{definition}

\begin{definition}[Comparison predicate]\label{def:pred_cmp}
    For expressions $a, b \in V$ and a comparison operator $\mathit{op} \in \{=,\, \neq,\, <,\, \leq,\, >,\, \geq\}$:
    \[
        \mathit{Cmp}(a,\, \mathit{op},\, b) \;=\; a \;\mathit{op}\; b
    \]
    The operators $<, \leq, >, \geq$ require $a, b$ to be from an ordered domain.
    Since $\Delta \subseteq \mathbb{R}$, durations are an ordered domain and duration comparison is a valid instance of $\mathit{Cmp}$.
\end{definition}

\begin{definition}[Set membership]\label{def:pred_contains}
    For a value $v \in V_{\text{nonset}}$ and a set value $s \in \mathcal{P}_{\text{fin}}(V_{\text{nonset}})$:
    \[
        \mathit{Contains}(v,\, s) \;\Leftrightarrow\; v \in s
    \]
\end{definition}

\begin{definition}[Negation]\label{def:pred_not}
    For a predicate $P \in \mathbf{Pred}$:
    \[
        \mathit{Not}(P) \;=\; \lnot P
    \]
\end{definition}

\begin{definition}[Conjunction and disjunction]\label{def:pred_allandany}
    For a finite set of predicates $C \subseteq \mathbf{Pred}$:
    \[
        \mathit{AllOf}(C) \;=\; \bigwedge_{P \in C} P \qquad
        \mathit{AnyOf}(C) \;=\; \bigvee_{P \in C} P
    \]
\end{definition}

\begin{definition}[Implication]\label{def:pred_implies}
    For predicates $A, B \in \mathbf{Pred}$:
    \[
        \mathit{Implies}(A,\, B) \;=\; A \Rightarrow B
    \]
    where $\Rightarrow$ is the standard logical implication.
\end{definition}

\begin{definition}[Quantifiers]\label{def:pred_quant}
    For a finite set $S$ and a predicate $P(x)$ parameterised by $x \in S$:
    \[
        \mathit{ForAll}(S,\, P) \;=\; \bigwedge_{x \in S} P(x) \qquad
        \mathit{Exists}(S,\, P) \;=\; \bigvee_{x \in S} P(x)
    \]
\end{definition}

\begin{definition}[Event happening]\label{def:pred_happening}
    For an entity $e \in E$ with $t_e = \text{Event}$ and a time point $t \in T$:
    \[
        \mathit{Happening}(e,\, t) \;\Leftrightarrow\; \mathit{Val}_\rho(e,\, t) = \mathit{true}
    \]
\end{definition}

\begin{definition}[Event has happened]\label{def:pred_has_happened}
    For an entity $e \in E$ with $t_e = \text{Event}$ and a time point $t \in T$:
    \[
        \mathit{HasHappened}_\rho(e,\, t) \;\Leftrightarrow\; \exists t' \leq t:\; \mathit{Happening}(e,\, t')
    \]
\end{definition}

The following predicates operate on intervals and durations.
Since $\Delta \subseteq \mathbb{R}$, duration comparison is a special case of $\mathit{Cmp}$ (Definition~\ref{def:pred_cmp}).

\begin{definition}[Interval inclusion]\label{def:pred_includes}
    Interval $i_1$ includes $i_2$ iff $i_2$ is entirely contained within $i_1$:
    \[
        \mathit{IntervalIncludes}(i_1,\, i_2) \;\Leftrightarrow\;
        \mathit{Start}(i_1) \leq \mathit{Start}(i_2) \;\land\; \mathit{End}(i_2) \leq \mathit{End}(i_1)
    \]
\end{definition}

\begin{definition}[Interval exclusion]\label{def:pred_excludes}
    Intervals $i_1$ and $i_2$ are exclusive iff they are completely disjoint:
    \[
        \mathit{IntervalExcludes}(i_1,\, i_2) \;\Leftrightarrow\;
        \mathit{End}(i_1) < \mathit{Start}(i_2) \;\lor\; \mathit{End}(i_2) < \mathit{Start}(i_1)
    \]
\end{definition}

\begin{definition}[Start included]\label{def:pred_start_in}
    The start of $i_1$ falls within $i_2$:
    \[
        \mathit{StartOfFirstIntervalIn}(i_1,\, i_2) \;\Leftrightarrow\;
        \mathit{Start}(i_2) \leq \mathit{Start}(i_1) \leq \mathit{End}(i_2)
    \]
\end{definition}

\begin{definition}[End included]\label{def:pred_end_in}
    The end of $i_1$ falls within $i_2$:
    \[
        \mathit{EndOfFirstIntervalIn}(i_1,\, i_2) \;\Leftrightarrow\;
        \mathit{Start}(i_2) \leq \mathit{End}(i_1) \leq \mathit{End}(i_2)
    \]
\end{definition}

\section{Operations}

An operation is an action that the system performs at a specific point in time.
Each operation has a set of arguments and produces \emph{effect predicates} — a formal statement, parameterised by the firing time $t$, of what must hold in the trace as a result of the operation being performed.
The operation is applied to some entity, which means that the argument list of every operation must include the entity the operation is applied to.

\begin{definition}[Operation]\label{def:operation}
    An \emph{operation} is a function
    \[
        op : T \times E \times A_1 \times \cdots \times A_n \rightarrow \mathbf{Pred}
    \]
    for some $n \geq 0$ and additional argument types $A_i \subseteq V \cup E$, where $T$ is the time set and $E$ is the mandatory target entity.
    The predicate $op(t,\, e,\, a_1,\, \ldots,\, a_n)$ is the \emph{effect} of $op$ applied to entity $e$ at time $t$ -- it describes the condition that the operation causes to hold.
    The trigger condition that invokes an operation is specified at the requirement level.
    We write $O$ for the finite set of all operations.
\end{definition}

\begin{definition}[Entity type operations function]\label{def:entity_type_ops_fn}
    \[
        O_T : T_E \rightarrow \mathcal{P}(O)
    \]
    maps each entity type and set of modifiers to its applicable operations.
\end{definition}

The concrete mapping, including argument types and effects, is as follows.
In effect predicates, $t_{\mathit{next}}$ denotes the next time \emph{any} value-modifying operation fires on entity $e$ after $t$, or $\infty$ if no such firing occurs:

\[
    t_{\mathit{next}} = \begin{cases}
        \min\{\, t' \in T \mid t' > t \;\land\; \text{a value-modifying } op' \in O_T(t_e) \text{ fires on } e \text{ at } t' \,\} & \text{if such } t' \text{ exists} \\
        \infty                                                                                                                     & \text{otherwise}
    \end{cases}
\]

, where $t$ is the time of the current firing and value-modifying operations are those whose effect predicate contains a $\mathit{Val}_\rho(e, -)$ term (i.e.\ $\mathit{calculate}$, $\mathit{write}$, $\mathit{set\_state}$, $\mathit{transmit}$, $\mathit{insert}$, $\mathit{remove}$, $\mathit{clear}$, and the destination side of $\mathit{read}$ and $\mathit{receive}$).

In the effect column, $\mathit{EvtPred}(e,\, ek,\, t)$ is used as shorthand for:
\[
    \forall\, ev \in E :\;
    \bigl(t_{ev} = \text{Event} \;\land\; \mu_{ev}(\mathit{target}) = id_e \;\land\; \mu_{ev}(\mathit{type}) = ek\bigr)
    \;\Rightarrow\; \mathit{Val}_\rho(ev,\, t) = \mathit{true}
\]
i.e.\ every \texttt{Event} entity declared with the matching $\mathit{target}$ and $\mathit{type}$ modifiers has its value set to $\mathit{true}$ at $t$, which by definition causes $\mathit{Happening}(ev,\, t)$ to hold.

\section{Temporal constructs}

So far, predicates ($\psi \in \mathbf{Pred}$) are point-in-time: they say whether something holds at a specific time $t$ in a specific trace $\rho$.
A \emph{trace predicate} lifts this to the whole trace: it says whether something holds across the entire execution, not just at one moment.
Formally, a trace predicate is a function $P : \Sigma^T \to \mathbb{B}$ — it takes a complete trace and returns true or false.
A trace $\rho$ \emph{satisfies} a requirement $R$ (written $\rho \models R$) iff $R$'s constraint, expressed as a trace predicate $P$, gives $P(\rho) = \mathit{true}$.

The temporal constructs below build trace predicates from point-in-time predicates $\psi \in \mathbf{Pred}$ by quantifying over time — expressing things like ``$\psi$ must hold at all times'' or ``whenever $\psi_{\mathit{cond}}$ holds, $\psi_{\mathit{effect}}$ must eventually follow.''

\begin{definition}[Trace predicate]\label{def:trace_predicate}
    A \emph{trace predicate} is a function $P : \Sigma^T \to \mathbb{B}$.
    We write $\mathbf{TracePred}$ for the set of all trace predicates.
    A trace $\rho$ \emph{satisfies} a trace predicate $P$ (written $\rho \models P$) iff $P(\rho) = \mathit{true}$.
\end{definition}

\begin{definition}[Temporal constructs]\label{def:temporal_constructs}
    \begin{align*}
        \mathit{Always}(\psi) \;                                                      & \Leftrightarrow\; \forall t \in T:\; \psi(\rho, t) \\
        \mathit{Eventually}(\psi) \;                                                  & \Leftrightarrow\; \exists t \in T:\; \psi(\rho, t) \\
        \mathit{Initial}(\psi) \;                                                     & \Leftrightarrow\; \psi(\rho, 0)                    \\
        \mathit{Causes}(\psi_{\mathit{cond}},\, \psi_{\mathit{effect}}) \;            & \Leftrightarrow\;
        \forall t \in T:\; \psi_{\mathit{cond}}(\rho, t) \Rightarrow \exists t' \geq t:\; \psi_{\mathit{effect}}(\rho, t')                 \\
        \mathit{CausesWithin}(\psi_{\mathit{cond}},\, \psi_{\mathit{effect}},\, d) \; & \Leftrightarrow\;
        \forall t \in T:\; \psi_{\mathit{cond}}(\rho, t) \Rightarrow \exists t' \in [t,\, t+d]:\; \psi_{\mathit{effect}}(\rho, t')         \\
        \mathit{Sequence}(\psi_1,\, \ldots,\, \psi_n) \;                              & \Leftrightarrow\;
        \exists t_1 \leq t_2 \leq \cdots \leq t_n \in T:\; \psi_i(\rho,\, t_i) \text{ for all } i                                          \\
        \mathit{Precedes}(\psi_1,\, \psi_2) \;                                        & \Leftrightarrow\;
        \forall t_1 \in T:\; \psi_2(\rho,\, t_1) \Rightarrow \exists t_2 \leq t_1:\; \psi_1(\rho,\, t_2)                                   \\
        \mathit{Excludes}(\psi_1,\, \psi_2) \;                                        & \Leftrightarrow\;
        \forall t \in T:\; \lnot\bigl(\psi_1(\rho,\, t) \land \psi_2(\rho,\, t)\bigr)                                                      \\
        \mathit{Immediately}(\psi_1,\, \psi_2) \;                                     & \Leftrightarrow\;
        \forall t_1 \in T:\; \psi_1(\rho,\, t_1) \Rightarrow \psi_2(\rho,\, \mathit{Next}(t_1))
    \end{align*}
    where $\psi,\, \psi_{\mathit{cond}},\, \psi_{\mathit{effect}},\, \psi_1, \ldots, \psi_n \in \mathbf{Pred}$ are point-in-time predicates and $d \in \Delta$ is a duration bound.
\end{definition}

The constructs compose via standard logical operators on trace predicates:
conjunction $P_1 \land P_2$, disjunction $P_1 \lor P_2$, negation $\lnot P$,
implication $P_1 \Rightarrow P_2$ (if $P_1$ holds for trace $\rho$ then $P_2$ must also hold), and
biconditional $P_1 \Leftrightarrow P_2$.
These operate on whole traces, not at individual time points.

\section{Requirements \& Test cases}

Both requirements and test cases operate over the same shared entity set $E$, which is what makes coverage checking possible.

A requirement is a formal constraint on the system's behaviour, expressed as a trace predicate.

\begin{definition}[Requirement]\label{def:requirement}
    A \emph{requirement} is a triple $R = (\mathit{flavour},\, \mathit{entities},\, \mathit{constraint})$ where:
    \begin{itemize}
        \item $\mathit{flavour} \in \{\text{Discrete},\, \text{Continuous}\}$ fixes the time set $T$ for this requirement ($T_D$ or $T_C$ respectively)
        \item $\mathit{entities} \subseteq E$ is the set of entities participating in this requirement
        \item $\mathit{constraint} \in \mathbf{TracePred}$ is the formal constraint on the trace
    \end{itemize}
    A trace $\rho$ \emph{satisfies} $R$ (written $\rho \models R$) iff $\mathit{constraint}(\rho) = \mathit{true}$.
\end{definition}

A test case exercises a subset of the same entities to produce a concrete trace, which the coverage checker then analyses against the requirement's constraint.

\begin{definition}[Stimulus]\label{def:stimulus}
    A \emph{stimulus} is a tuple $(t,\, op,\, e,\, a_1,\, \ldots,\, a_n)$ where:
    \begin{itemize}
        \item $t \in T$ is the firing time
        \item $op \in O$ is an operation
        \item $e \in E$ is the target entity
        \item $a_1, \ldots, a_n \in V \cup E$ are the additional arguments required by $op$
    \end{itemize}
    We write $\mathbf{Stim}$ for the set of all well-typed stimuli.
\end{definition}

\begin{definition}[Test case]\label{def:test_case}
    A \emph{test case} is a triple $\mathit{TestCase} = (\mathit{setup},\, \mathit{stimuli},\, \mathit{assertions})$ where:
    \begin{itemize}
        \item $\mathit{setup}$ is a valuation $\sigma_0 : E \rightharpoonup V$ establishing the initial state of the relevant entities
        \item $\mathit{stimuli} \in \mathcal{P}_{\mathrm{fin}}(\mathbf{Stim})$ is a finite set of stimuli
        \item $\mathit{assertions} : T \to \mathcal{P}(\mathbf{Pred})$ maps each time point to the set of point-in-time predicates the test verifies at that time; it has finite support
    \end{itemize}
\end{definition}
