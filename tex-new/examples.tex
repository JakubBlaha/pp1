\section{Examples}

The examples in this section are sample requirements provided by the stakeholder rewritten into a formal form using an Claude Sonnet 4.6 with prompting guidance. That means the examples were not one-shotted. It seems so far that the model is not capable of one shotting these requirements.

\subsection{Custom Counter}

\textbf{Requirement:} After the system enters emergency mode, at each discrete time step the counter \textit{counter\_1} is incremented by 3.

\medskip

\begin{center}
\begin{tabular}{lll}
    \toprule
    \textbf{Entity} & $t_e$ & $\mu_e$ \\
    \midrule
    $\mathit{enter\_emergency}$ & EventTrigger & --- \\
    $\mathit{ev\_enter\_emergency}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{enter\_emergency}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    $\mathit{counter\_1}$ & Signal & --- \\
    \bottomrule
\end{tabular}
\end{center}

\textbf{Constraint:}
\begin{align*}
    \mathit{Always}\Bigl(\;
        \mathit{HasHappened}_\rho(\mathit{ev\_enter\_emergency},\, \mathit{Now})
        \Rightarrow \mathit{Val}_\rho(\mathit{counter\_1},\, \mathit{Now}) = \mathit{Val}_\rho(\mathit{counter\_1},\, \mathit{Prev}(\mathit{Now})) + 3
    \;\Bigr)
\end{align*}

The antecedent holds at time $t$ iff $\mathit{enter\_emergency}$ has happened at least once up to $t$.
The consequent requires $\mathit{counter\_1}$ to equal its previous value plus 3 at every such time step.

\medskip

\begin{center}
\begin{tabular}{ll}
    \toprule
    \textbf{Entity} & $t_e$ \\
    \midrule
    $\mathit{EngagementNoSat\_u}$ & Signal \\
    $\mathit{Switch\_b}$ & Signal \\
    $\mathit{Lon\_u}$ & Signal \\
    \bottomrule
\end{tabular}
\end{center}

\textbf{Flavour:} Discrete.

\textbf{Constraint:}
\begin{align*}
    &\mathit{Initial}\bigl(\mathit{Val}_\rho(\mathit{EngagementNoSat\_u},\, 0) = 0\bigr) \\
    \land\; &\mathit{Always}\Bigl(\\
    &\quad \bigl(\mathit{Val}_\rho(\mathit{Switch\_b},\, \mathit{Now}) = \mathit{true}\bigr) \Rightarrow {} \\
    &\qquad \mathit{Val}_\rho(\mathit{EngagementNoSat\_u},\, \mathit{Now}) = 1.2 \cdot \mathit{Val}_\rho(\mathit{EngagementNoSat\_u},\, \mathit{Prev}(\mathit{Now}))\\
    &\land\; \bigl(\mathit{Val}_\rho(\mathit{Switch\_b},\, \mathit{Now}) \neq \mathit{true}\bigr) \Rightarrow {} \\
    &\qquad \mathit{Val}_\rho(\mathit{EngagementNoSat\_u},\, \mathit{Now}) = \mathit{Val}_\rho(\mathit{Lon\_u},\, \mathit{Now})
    \;\Bigr)
\end{align*}

\subsection{Req 05 -- Exception handling procedure}

\textbf{Requirement:} The software, when handling an exception, must execute the following steps in order:
(1) copy register SRR0 to R5;
(2) invoke the image exception handler;
(3) if the handler returns, hold software execution.

\medskip
\begin{center}
\begin{tabular}{lll}
    \toprule
    \textbf{Entity} & $t_e$ & $\mu_e$ \\
    \midrule
    $\mathit{SRR0}$ & Storage & $\{\mathit{register} \mapsto \top\}$ \\
    $\mathit{R5}$ & Storage & $\{\mathit{register} \mapsto \top\}$ \\
    $\mathit{ev\_written\_r5}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{R5}},\ \mathit{type} \mapsto \mathit{written}\}$ \\
    $\mathit{exception}$ & EventTrigger & --- \\
    $\mathit{ev\_exception}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{exception}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    $\mathit{invoke\_ieh}$ & EventTrigger & --- \\
    $\mathit{ev\_invoke\_ieh}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{invoke\_ieh}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    $\mathit{return\_ieh}$ & EventTrigger & --- \\
    $\mathit{ev\_return\_ieh}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{return\_ieh}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    $\mathit{hold\_exec}$ & EventTrigger & --- \\
    $\mathit{ev\_hold\_exec}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{hold\_exec}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    \bottomrule
\end{tabular}
\end{center}

\textbf{Flavour:} Discrete.

\textbf{Constraint:}
\begin{align*}
    &\mathit{Sequence}\Bigl( \\
    &\quad \mathit{Happening}(\mathit{ev\_exception},\, \mathit{Now}), \\
    &\quad \mathit{Happening}(\mathit{ev\_written\_r5},\, \mathit{Now}) \\
    &\qquad \land\; \mathit{Val}_\rho(\mathit{R5},\, \mathit{Now}) = \mathit{Val}_\rho(\mathit{SRR0},\, \mathit{Now}), \\
    &\quad \mathit{Happening}(\mathit{ev\_invoke\_ieh},\, \mathit{Now})
    \;\Bigr) \\
    \land\; &\mathit{Causes}\Bigl(
        \mathit{Happening}(\mathit{ev\_return\_ieh},\, \mathit{Now}),\;
        \mathit{Happening}(\mathit{ev\_hold\_exec},\, \mathit{Now})
    \Bigr)
\end{align*}

$\mathit{Sequence}$ is existential: it requires at least one complete chain to occur in the trace.
The universal reading --- every exception must trigger this sequence --- would require a triggered-sequence construct not yet in the formalism.
The $\mathit{Causes}$ clause is conditional: it fires only if the handler returns, directly modelling the ``if returns'' branch.

\textbf{Requirement:} The software must start executing the sequence of tasks allocated to the execution schedule in less than 2 seconds after processor power-up.

\medskip

\begin{tabular}{lll}

    \textbf{Entity} & $t_e$ & $\mu_e$ \\

    $\mathit{processor\_power\_up}$ & EventTrigger & --- \\
    $\mathit{ev\_power\_up}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{processor\_power\_up}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    $\mathit{invoke\_task\_seq}$ & EventTrigger & --- \\

\end{tabular}
\end{center}

\textbf{Flavour:} Continuous (time unit: seconds).

\textbf{Constraint:}
\[
    \mathit{CausesWithin}\Bigl(

\subsection{Req 09 -- Timed signal toggle with multi-condition guard}

\textbf{Requirement:} The software must toggle \textit{valid\_range}, within 500\,ns of a write to $d$, if ALL the following conditions are satisfied:
\begin{itemize}
    \item The difference between last Maximum Peak and Minimum Peak of $d$ is higher than 655.
    \item It is the second write to $d$ after the Minimum Peak detection.
    \item Signed value of $d$ is higher than the signed value of last $d$.
\end{itemize}

\medskip
\textbf{Entities:}
\begin{center}
\begin{tabular}{lll}
    \toprule
    \textbf{Entity} & $t_e$ & $\mu_e$ \\
    \midrule
    $d$ & Signal & --- \\
    $\mathit{valid\_range}$ & Signal & --- \\
    $\mathit{ev\_written\_d}$ & Event & $\{\mathit{target} \mapsto id_d,\ \mathit{type} \mapsto \mathit{written}\}$ \\
    $\mathit{ev\_written\_vr}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{valid\_range}},\ \mathit{type} \mapsto \mathit{written}\}$ \\
    $\mathit{min\_peak\_det}$ & EventTrigger & --- \\
    $\mathit{ev\_min\_peak\_det}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{min\_peak\_det}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    $\mathit{max\_peak\_det}$ & EventTrigger & --- \\
    $\mathit{ev\_max\_peak\_det}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{max\_peak\_det}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    \bottomrule
\end{tabular}
\end{center}

\textbf{Flavour:} Continuous (time unit: nanoseconds).

\medskip
Let $t_{\mathit{mpd}} = \mathit{Start}(\mathit{LastOcc}_\rho(\mathit{ev\_min\_peak\_det},\, \mathit{Now},\, 1))$,
$t_{\mathit{Mpd}} = \mathit{Start}(\mathit{LastOcc}_\rho(\mathit{ev\_max\_peak\_det},\, \mathit{Now},\, 1))$, and
$t_{\mathit{prevd}} = \mathit{Start}(\mathit{LastOcc}_\rho(\mathit{ev\_written\_d},\, \mathit{Now},\, 2))$
abbreviate the times of the most recent minimum peak detection, maximum peak detection, and second-to-last write to $d$, all up to $\mathit{Now}$.

\textbf{Constraint:}
\begin{align*}
    &\mathit{CausesWithin}\Bigl( \\
    &\quad \mathit{Happening}(\mathit{ev\_written\_d},\, \mathit{Now}) \\
    &\quad \land\; \mathit{Val}_\rho(d,\, t_{\mathit{Mpd}}) - \mathit{Val}_\rho(d,\, t_{\mathit{mpd}}) > 655 \\
    &\quad \land\; \mathit{EvtOccCount}_\rho(\mathit{ev\_written\_d},\, \mathit{MkIntervalOC}(t_{\mathit{mpd}},\, \mathit{Now})) = 2 \\
    &\quad \land\; \mathit{Val}_\rho(d,\, \mathit{Now}) > \mathit{Val}_\rho(d,\, t_{\mathit{prevd}}), \\
    &\quad \mathit{Happening}(\mathit{ev\_written\_vr},\, \mathit{Now}) \\
    &\qquad \land\; \mathit{Val}_\rho(\mathit{valid\_range},\, \mathit{Now}) = \lnot\mathit{ValBefore}_\rho(\mathit{valid\_range},\, \mathit{Now}), \\
    &\quad 500
    \;\Bigr)
\end{align*}

\textit{Condition 2}: $\mathit{MkIntervalOC}(t_{\mathit{mpd}},\, \mathit{Now})$ is open on the left, so a write coinciding exactly with the minimum peak detection at $t_{\mathit{mpd}}$ is not counted, correctly capturing ``after the Minimum Peak detection''.
\textit{Condition 3}: ``signed value of last $d$'' means the value of $d$ at the previous write event, captured explicitly by $\mathit{Val}_\rho(d,\, t_{\mathit{prevd}})$. The signed integer interpretation has no counterpart in the formalism since $d$ is treated as a real-valued signal.
$\mathit{valid\_range}$ is assumed boolean; $\lnot$ denotes boolean negation.

\subsection{Req 13 -- Bounded count of large register values}

\textbf{Requirement:} The number of values larger than 10 in the four registers A, B, C, D when they are read at the same timestep is never greater than 2.

\medskip
\begin{center}
\begin{tabular}{lll}
    \toprule
    \textbf{Entity} & $t_e$ & $\mu_e$ \\
    \midrule
    $A$ & Storage & $\{\mathit{register} \mapsto \top\}$ \\
    $B$ & Storage & $\{\mathit{register} \mapsto \top\}$ \\
    $C$ & Storage & $\{\mathit{register} \mapsto \top\}$ \\
    $D$ & Storage & $\{\mathit{register} \mapsto \top\}$ \\
    $\mathit{ev\_read\_A}$ & Event & $\{\mathit{target} \mapsto id_A,\ \mathit{type} \mapsto \mathit{read}\}$ \\
    $\mathit{ev\_read\_B}$ & Event & $\{\mathit{target} \mapsto id_B,\ \mathit{type} \mapsto \mathit{read}\}$ \\
    $\mathit{ev\_read\_C}$ & Event & $\{\mathit{target} \mapsto id_C,\ \mathit{type} \mapsto \mathit{read}\}$ \\
    $\mathit{ev\_read\_D}$ & Event & $\{\mathit{target} \mapsto id_D,\ \mathit{type} \mapsto \mathit{read}\}$ \\
    \bottomrule
\end{tabular}
\end{center}

\textbf{Flavour:} Discrete.

\textbf{Constraint:}

Let $\mathit{regs} = \{A,\, B,\, C,\, D\} \in \mathcal{P}_{\text{fin}}(V_{\text{nonset}})$ be the set of register entities.
Both the antecedent and the consequent quantify over $\mathit{regs}$:

\begin{align*}
    \mathit{Always}\Bigl(
        &\mathit{ForAll}\bigl(\mathit{regs},\;
            \lambda e.\; \mathit{Happening}(e,\, \mathit{read},\, \mathit{Now})\bigr) \\
        &\Rightarrow \mathit{Cmp}\Bigl(
            \mathit{Size}\bigl(
                \mathit{Filter}\bigl(\mathit{regs},\;
                    \lambda e.\; \mathit{Val}_\rho(e,\, \mathit{Now}) > 10\bigr)
            \bigr),\; \leq,\; 2
        \Bigr)
    \Bigr)
\end{align*}

The antecedent holds at time $t$ iff all four registers are read simultaneously.
The consequent builds the set of registers whose value exceeds 10 and asserts its cardinality is at most 2.

\subsection{Req 10 -- Repeated failure triggers emergency mode}

\textbf{Requirement:} The software shall enter the emergency mode if any of the following failures individually occurs twice in less than 10 seconds:
\begin{itemize}
    \item[(a)] sensor failure
    \item[(b)] energy failure
    \item[(c)] communication failure.
\end{itemize}

\medskip
\textbf{Entities:}
\begin{center}
\begin{tabular}{lll}
    \toprule
    \textbf{Entity} & $t_e$ & $\mu_e$ \\
    \midrule
    $\mathit{sensor\_failure}$  & EventTrigger & --- \\
    $\mathit{ev\_sensor\_fail}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{sensor\_failure}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    $\mathit{energy\_failure}$  & EventTrigger & --- \\
    $\mathit{ev\_energy\_fail}$ & Event & $\{\mathit{target} \mapsto id_{\mathit{energy\_failure}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    $\mathit{comm\_failure}$    & EventTrigger & --- \\
    $\mathit{ev\_comm\_fail}$   & Event & $\{\mathit{target} \mapsto id_{\mathit{comm\_failure}},\ \mathit{type} \mapsto \mathit{generic}\}$ \\
    $\mathit{emergency\_mode}$  & State & --- \\
    \bottomrule
\end{tabular}
\end{center}

\textbf{Flavour:} Continuous (time unit: seconds).

\textbf{Constraint:}
\begin{align*}
    \mathit{Causes}\Bigl(
        &\;\Bigl(\mathit{Happening}(\mathit{ev\_sensor\_fail},\, \mathit{Now}) \\
        &\qquad\land\; \mathit{EvtOccCount}_\rho\bigl(\mathit{ev\_sensor\_fail},\, \mathit{MkIntervalOO}(\mathit{Now} - 10,\, \mathit{Now})\bigr) \geq 1\Bigr) \\
        \lor\;
        &\;\Bigl(\mathit{Happening}(\mathit{ev\_energy\_fail},\, \mathit{Now}) \\
        &\qquad\land\; \mathit{EvtOccCount}_\rho\bigl(\mathit{ev\_energy\_fail},\, \mathit{MkIntervalOO}(\mathit{Now} - 10,\, \mathit{Now})\bigr) \geq 1\Bigr) \\
        \lor\;
        &\;\Bigl(\mathit{Happening}(\mathit{ev\_comm\_fail},\, \mathit{Now}) \\
        &\qquad\land\; \mathit{EvtOccCount}_\rho\bigl(\mathit{ev\_comm\_fail},\, \mathit{MkIntervalOO}(\mathit{Now} - 10,\, \mathit{Now})\bigr) \geq 1\Bigr), \\
        &\;\mathit{Val}_\rho(\mathit{emergency\_mode},\, \mathit{Now}) = \mathit{true}
    \;\Bigr)
\end{align*}

\textit{``Individually''}: the three failure types are tracked separately; $\psi$ is instantiated independently for each event entity, so a sensor failure and an energy failure occurring within 10 seconds of each other do not jointly trigger the condition.
\textit{``Twice in less than 10 seconds''}: $\mathit{MkIntervalOO}(\mathit{Now} - 10,\, \mathit{Now})$ is open on both ends: the left endpoint excludes a gap of exactly 10 seconds (``less than''), and the right endpoint excludes $\mathit{Now}$ itself so that only the previous occurrence is counted alongside the current $\mathit{Happening}$.
\textit{Edge case}: when $\mathit{Now} < 10$, the expression $\mathit{Now} - 10$ is negative and outside $T = \mathbb{R}_{\geq 0}$; in practice no event can occur before time 0, so the count is still correct, but a well-formed guard $\mathit{Now} \geq 10$ would be needed to make the interval argument a valid element of $T$.
